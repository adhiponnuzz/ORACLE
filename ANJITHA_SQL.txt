(2)
SELECT NAME,TOTAL_MARKS FROM(SELECT NAME,TOTAL_MARKS FROM STUDENT ORDER BY TOTAL_MARKS ASC)WHERE ROWNUM<=3;

(3)
SELECT SUBJECT_NAME FROM SUBJECT S LEFT JOIN TEACHER_TABLE T ON S.SUBJ_ID=T.SUBJ_ID WHERE T.SUBJ_ID IS NULL;

(4)
SELECT R.ADMNO,NAME,R.SUBJ_ID FROM RESULTS R JOIN STUDENT S ON R.ADMNO=S.ADMNO WHERE(R.SUBJ_ID,R.MARKS )IN (SELECT SUBJ_ID, MAX(MARKS) 
FROM RESULTS
GROUP BY SUBJ_ID);

(5)
